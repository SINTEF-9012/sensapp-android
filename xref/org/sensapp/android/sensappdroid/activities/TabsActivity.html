71      public static class TabManager implements TabHost.OnTabChangeListener {
72          
73      	private final FragmentActivity tabActivity;
74          private final TabHost tabHost;
75          private final int containerId;
76          private final HashMap<String, TabInfo> tabs = new HashMap<String, TabInfo>();
77          private TabInfo lastTab;
78  
79          static final class TabInfo {
80              private final String tag;
81              private final Class<?> clss;
82              private final Bundle args;
83              private Fragment fragment;
84              TabInfo(String tag, Class<?> clss, Bundle args) {
85                  this.tag = tag;
86                  this.clss = clss;
87                  this.args = args;
88              }
89          }
90  
91          static class DummyTabFactory implements TabHost.TabContentFactory {
92          	private final Context context;
93              public DummyTabFactory(Context context) {
94                  this.context = context;
95              }
96              @Override
97              public View createTabContent(String tag) {
98                  View v = new View(context);
99                  v.setMinimumWidth(0);
100                 v.setMinimumHeight(0);
101                 return v;
102             }
103         }
104 
105         public TabManager(FragmentActivity activity, TabHost tabHost, int containerId) {
106             this.tabActivity = activity;
107             this.tabHost = tabHost;
108             this.containerId = containerId;
109             this.tabHost.setOnTabChangedListener(this);
110         }
111 
112         public void addTab(TabHost.TabSpec tabSpec, Class<?> clss, Bundle args) {
113             tabSpec.setContent(new DummyTabFactory(tabActivity));
114             String tag = tabSpec.getTag();
115 
116             TabInfo info = new TabInfo(tag, clss, args);
117             info.fragment = tabActivity.getSupportFragmentManager().findFragmentByTag(tag);
118 
119             // Check to see if we already have a fragment for this tab, probably
120             // from a previously saved state.  If so, deactivate it, because our
121             // initial state is that a tab isn't shown.
122             if (info.fragment != null && !info.fragment.isDetached()) {
123                 FragmentTransaction ft = tabActivity.getSupportFragmentManager().beginTransaction();
124                 ft.detach(info.fragment);
125                 ft.commit();
126             }
127 
128             tabs.put(tag, info);
129             tabHost.addTab(tabSpec);
130         }
131 
132         @Override
133         public void onTabChanged(String tabId) {
134             TabInfo newTab = tabs.get(tabId);
135             if (lastTab != newTab) {
136                 FragmentTransaction ft = tabActivity.getSupportFragmentManager().beginTransaction();
137                 if (lastTab != null && lastTab.fragment != null) {
138                         ft.detach(lastTab.fragment);
139                 }
140                 if (newTab != null) {
141                     if (newTab.fragment == null) {
142                         newTab.fragment = Fragment.instantiate(tabActivity, newTab.clss.getName(), newTab.args);
143                         ft.add(containerId, newTab.fragment, newTab.tag);
144                     } else {
145                         ft.attach(newTab.fragment);
146                     }
147                 }
148 
149                 lastTab = newTab;
150                 ft.commit();
151                 tabActivity.getFragmentManager().executePendingTransactions();
152             }
153         }
154     }
