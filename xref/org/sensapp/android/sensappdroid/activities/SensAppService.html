19  import java.util.Set;
20  
21  import org.sensapp.android.sensappdroid.connectivity.ConnectivityReceiver;
22  import org.sensapp.android.sensappdroid.contract.SensAppContract;
23  import org.sensapp.android.sensappdroid.datarequests.DeleteMeasuresTask;
24  import org.sensapp.android.sensappdroid.preferences.AutoUploadSensorDialog;

28  import android.app.Service;
29  import android.content.Intent;
30  import android.net.Uri;
31  import android.os.Bundle;
32  import android.os.IBinder;
33  import android.preference.PreferenceManager;
34  import android.util.Log;
35  import android.widget.Toast;
36  
37  public class SensAppService extends Service implements PutMeasureCallback {
38  	
39  	public static final String ACTION_UPLOAD = SensAppService.class.getName() + ".ACTION_UPLOAD";
40  	public static final String ACTION_AUTO_UPLOAD = SensAppService.class.getName() + ".ACTION_AUTO_UPLOAD";
41  	public static final String ACTION_DELETE_LOCAL = SensAppService.class.getName() + ".ACTION_DELETE_LOCAL";
42  	public static final String EXTRA_UPLOADED_FILTER = SensAppService.class.getName() + ".EXTRA_UPLOADED_FILTER";
43  	
44  	private static final String TAG = SensAppService.class.getSimpleName();
45  	
46  	private boolean waitForData = false;
47  	
48  	private int lastTaskStarted = 0;
49  	private int lastConsecutiveTaskEnded = 0;
50  	private ArrayList<Integer> taskBuffer = new ArrayList<Integer>();
51  	
52  	@Override
53  	public void onCreate() {
54  		super.onCreate();
55  	}
56  	
57  	@Override
58  	public int onStartCommand(Intent intent, int flags, int startId) {
59  		if (intent.getAction() != null) {
60  			if (intent.getAction().equals(ACTION_UPLOAD)) {
61  				Log.d(TAG, "Receive: ACTION_UPLOAD");
62  				uploadMeasureUri(intent.getData());
63  			} else if (intent.getAction().equals(ACTION_AUTO_UPLOAD)) {
64  				Log.d(TAG, "Receive: ACTION_AUTO_UPLOAD");
65  				autoUpload();
66  			} else if (intent.getAction().equals(ACTION_DELETE_LOCAL)) {
67  				Log.d(TAG, "Receive: ACTION_DELETE_LOCAL");
68  				Bundle extra = intent.getExtras();
69  				if (extra == null) {
70  					new DeleteMeasuresTask(this, intent.getData()).execute();
71  				} else if (extra.getBoolean(EXTRA_UPLOADED_FILTER)) {
72  					new DeleteMeasuresTask(this, intent.getData()).execute(SensAppContract.Measure.UPLOADED + " = 1");
73  				}
74  			} else if (intent.getAction().equals(ConnectivityReceiver.ACTION_CONNECTIVITY_FOUND)) {
75  				Log.d(TAG, "Receive: ACTION_DELETE_LOCAL");
76  				if (waitForData) {
77  					autoUpload();
78  				} else {
79  					stopSelf();
80  				}
81  			}
82  		} else {
83  			stopSelf();
84  		}
85  		return START_NOT_STICKY;
86  	}
87  	
88  	private void uploadMeasureUri(Uri uri) {
89  		if (!ConnectivityReceiver.isDataAvailable(getApplicationContext())) {
90  			Log.e(TAG, "No data connection available");
91  			Toast.makeText(this, "No data connection available", Toast.LENGTH_LONG).show();
92  		} else {
93  			new PutMeasuresTask(this, taskIdGen(), getApplicationContext(), uri).execute();
94  		}
95  	}
96  	
97  	private void autoUpload() {
98  		if (!ConnectivityReceiver.isDataAvailable(getApplicationContext())) {
99  			Log.e(TAG, "No data connection available");
100 			waitForData = true;
101 		} else {
102 			waitForData = false;
103 			Set<String> names = PreferenceManager.getDefaultSharedPreferences(this).getStringSet(AutoUploadSensorDialog.SENSOR_MAINTAINED, null);
104 			if (names != null && !names.isEmpty()) {
105 				for (final String name : names) {
106 					Log.d(TAG, "Put " + name + " sensor (Auto upload)");
107 					new PutMeasuresTask(this, taskIdGen(), getApplicationContext(), Uri.parse(SensAppContract.Measure.CONTENT_URI + "/" + name), PutMeasuresTask.FLAG_SILENT).execute();
108 				}
109 			}
110 		}
111 	}
112 	
113 	private int taskIdGen() {
114 		return ++ lastTaskStarted;
115 	}
116 	
117 	@Override
118 	public IBinder onBind(Intent intent) {
119 		return null;
120 	}
121 
122 	@Override
123 	public void onTaskFinished(int id) {
124 		Log.e(TAG, "task finished: " + id + ", laststarted: " + lastTaskStarted + ", lastEnded: " + lastConsecutiveTaskEnded);
125 		taskBuffer.add(id);
126 		while (taskBuffer.contains(lastConsecutiveTaskEnded + 1)) {
127 			lastConsecutiveTaskEnded ++;
128 			taskBuffer.remove((Integer) lastConsecutiveTaskEnded);
129 		}
130 		if (lastTaskStarted == lastConsecutiveTaskEnded) {
131 			Log.e(TAG, "__FINISH__");
132 			stopSelf();
133 		}
134 	}
