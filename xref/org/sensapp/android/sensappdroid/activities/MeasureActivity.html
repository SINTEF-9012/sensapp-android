16  package org.sensapp.android.sensappdroid.activities;
17  
18  import org.sensapp.android.sensappdroid.R;
19  import org.sensapp.android.sensappdroid.contract.SensAppContract;
20  import org.sensapp.android.sensappdroid.database.MeasureTable;
21  
22  import android.app.Activity;
23  import android.database.Cursor;
24  import android.net.Uri;
25  import android.os.Bundle;
26  import android.widget.TextView;
27  
28  public class MeasureActivity extends Activity {
29  
30  	private TextView tvId;
31  	private TextView tvSensor;
32  	private TextView tvValue;
33  	private TextView tvTime;
34  	private TextView tvStatus;
35  	
36  	@Override
37  	protected void onCreate(Bundle savedInstanceState) {
38  		super.onCreate(savedInstanceState);
39  		setContentView(R.layout.measure_view);
40  		tvId = (TextView) findViewById(R.id.measure_id_value);
41  		tvSensor = (TextView) findViewById(R.id.measure_sensor_value);
42  		tvValue = (TextView) findViewById(R.id.measure_value_value);
43  		tvTime = (TextView) findViewById(R.id.measure_time_value);
44  		tvStatus = (TextView) findViewById(R.id.measure_status_value);
45  		Uri uri = getIntent().getData();		
46  		fillData(uri);
47  	}
48  
49  	private void fillData(Uri uri) {
50  		Cursor cursor = getContentResolver().query(uri, null, null, null, null);
51  		if (cursor != null) {
52  			cursor.moveToFirst();
53  			tvId.setText(cursor.getString(cursor.getColumnIndexOrThrow(SensAppContract.Measure.ID)));
54  			tvSensor.setText(cursor.getString(cursor.getColumnIndexOrThrow(SensAppContract.Measure.SENSOR)));
55  			tvValue.setText(cursor.getString(cursor.getColumnIndexOrThrow(SensAppContract.Measure.VALUE)));
56  			tvTime.setText(cursor.getString(cursor.getColumnIndexOrThrow(MeasureTable.COLUMN_TIME)));
57  			if (cursor.getInt(cursor.getColumnIndexOrThrow(MeasureTable.COLUMN_UPLOADED)) != 0) {
58  				tvStatus.setText(R.string.measure_uploaded);
59  			} else {
60  				tvStatus.setText(R.string.measure_notuploaded);
61  			}
62  			cursor.close();
63  		}
64  	}
65  }
