16  package org.sensapp.android.sensappdroid.database;
17  
18  import android.content.Context;
19  import android.database.sqlite.SQLiteDatabase;
20  import android.database.sqlite.SQLiteOpenHelper;
21  import android.util.Log;
22  
23  public class SensAppDatabaseHelper extends SQLiteOpenHelper {
24  	
25  	private static final String TAG = SensAppDatabaseHelper.class.getName();
26  	private static final String DATABASE_NAME = "sensapp.db";
27  	private static final int DATABASE_VERSION = 14;
28  	
29  	public SensAppDatabaseHelper(Context context) {
30  		super(context, DATABASE_NAME, null, DATABASE_VERSION);
31  	}
32  	
33  	@Override
34  	public void onCreate(SQLiteDatabase database) {
35  		Log.i(TAG, "Creating DATABASE " + DATABASE_NAME);
36  		MeasureTable.onCreate(database);
37  		SensorTable.onCreate(database);
38  		CompositeTable.onCreate(database);
39  		ComposeTable.onCreate(database);
40  		//MetadataTable.onCreate(database);
41  	}
42  
43  	@Override
44  	public void onUpgrade(SQLiteDatabase database, int oldVersion, int newVersion) {
45  		Log.i(TAG, "Updating DATABASE " + DATABASE_NAME + " from " + oldVersion + " to " + newVersion);
46  		MeasureTable.onUpgrade(database, oldVersion, newVersion);
47  		SensorTable.onUpgrade(database, oldVersion, newVersion);
48  		CompositeTable.onUpgrade(database, oldVersion, newVersion);
49  		ComposeTable.onUpgrade(database, oldVersion, newVersion);
50  		//MetadataTable.onUpgrade(database, oldVersion, newVersion);
51  	}
52  }
