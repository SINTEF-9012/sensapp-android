100 	private boolean sensorExists(String sensorName) {
101 		String[] projection = {SensAppContract.Sensor.NAME};
102 		Cursor cursor = context.getContentResolver().query(Uri.parse(SensAppContract.Sensor.CONTENT_URI + "/" + sensorName), projection, null, null, null); 
103 		if (cursor != null) {
104 			boolean exists =  cursor.getCount() > 0;
105 			cursor.close();
106 			return exists;
107 		}
108 		return false;
109 	}

111 	private List<Long> getBasetimes(String sensorName) {
112 		List<Long> basetimes = new ArrayList<Long>();
113 		String[] projection = {"DISTINCT " + SensAppContract.Measure.BASETIME};
114 		Cursor cursor = context.getContentResolver().query(Uri.parse(SensAppContract.Measure.CONTENT_URI + "/" + sensorName), projection, null, null, null);
115 		if (cursor != null) {
116 			while (cursor.moveToNext()) {
117 				basetimes.add(cursor.getLong(cursor.getColumnIndexOrThrow(SensAppContract.Measure.BASETIME)));
118 			}
119 			cursor.close();
120 		}
121 		return basetimes;
122 	}

139 	@Override
140 	protected Integer doInBackground(Integer... params) {
141 
142 		int rowTotal = 0;
143 
144 		Cursor cursor = context.getContentResolver().query(uri, new String[]{SensAppContract.Measure.ID}, SensAppContract.Measure.UPLOADED + " = 0", null, null);
145 		if (cursor != null) {
146 			rowTotal = cursor.getCount();
147 			cursor.close();
148 		}
149 
150 		if (flag != FLAG_SILENT) {
151 			notification.contentView.setTextViewText(R.id.status_text, "Uploading " + rowTotal + " measures...");
152 			notificationManager.notify(NOTIFICATION_ID, notification);
153 		}
154 		
155 		int rowsUploaded = 0;
156 		int progress = 0;
157 		int sizeLimit = DEFAULT_SIZE_LIMIT;
158 		if (params.length > 0) {
159 			sizeLimit = params[0];
160 		}
161 
162 		ArrayList<String> sensorNames = new ArrayList<String>();
163 		cursor = context.getContentResolver().query(uri, new String[]{"DISTINCT " + SensAppContract.Measure.SENSOR}, SensAppContract.Measure.UPLOADED + " = 0", null, null);
164 		if (cursor != null) {
165 			while (cursor.moveToNext()) {
166 				sensorNames.add(cursor.getString(cursor.getColumnIndexOrThrow(SensAppContract.Measure.SENSOR)));
167 			}
168 			cursor.close();
169 		}
170 
171 		Sensor sensor;
172 		for (String sensorName : sensorNames) {
173 
174 			if (!sensorExists(sensorName)) {
175 				Log.e(TAG, "Incorrect database: sensor " + sensorName + " does not exit");
176 				return null;
177 			}
178 
179 			// Update uri with current preference
180 			try {
181 				ContentValues values = new ContentValues();
182 				values.put(SensAppContract.Sensor.URI, GeneralPrefFragment.buildUri(PreferenceManager.getDefaultSharedPreferences(context), context.getResources()));
183 				context.getContentResolver().update(Uri.parse(SensAppContract.Sensor.CONTENT_URI + "/" + sensorName), values, null, null);
184 			} catch (IllegalStateException e) {
185 				errorMessage = e.getMessage();
186 				Log.e(TAG, errorMessage);
187 				return null;
188 			}
189 
190 			sensor = DatabaseRequest.SensorRQ.getSensor(context, sensorName);
191 
192 			try {
193 				if (!RestRequest.isSensorRegistred(sensor)) {
194 					Uri postSensorResult = null;
195 					try {
196 						postSensorResult = new PostSensorRestTask(context, sensorName).executeOnExecutor(THREAD_POOL_EXECUTOR).get();
197 					} catch (InterruptedException e) {
198 						e.printStackTrace();
199 					} catch (ExecutionException e) {
200 						e.printStackTrace();
201 					}
202 					if (postSensorResult == null) {
203 						Log.e(TAG, "Post sensor failed");
204 						return null;
205 					}
206 				}
207 			} catch (RequestErrorException e1) {
208 				errorMessage = e1.getMessage();
209 				Log.e(TAG, errorMessage);
210 				return null;
211 			}
212 
213 			MeasureJsonModel model = null;
214 			if ("Numerical".equals(sensor.getTemplate())) {
215 				model = new NumericalMeasureJsonModel(sensorName, sensor.getUnit());
216 			} else if ("String".equals(sensor.getTemplate())) {
217 				model = new StringMeasureJsonModel(sensorName, sensor.getUnit());
218 			} else {
219 				errorMessage = "Incorrect sensor template";
220 				Log.e(TAG, errorMessage);
221 				return null;
222 			}
223 
224 			List<Integer> ids = new ArrayList<Integer>();
225 			for (Long basetime : getBasetimes(sensorName)) {
226 				model.setBt(basetime);	
227 				String[] projection = {SensAppContract.Measure.ID, SensAppContract.Measure.VALUE, SensAppContract.Measure.TIME};
228 				String selection = SensAppContract.Measure.SENSOR + " = \"" + model.getBn() + "\" AND " + SensAppContract.Measure.BASETIME + " = " + model.getBt() + " AND " + SensAppContract.Measure.UPLOADED + " = 0";
229 				cursor = context.getContentResolver().query(uri, projection, selection, null, null);
230 				if (cursor != null) {
231 					if (cursor.getCount() > 0) {
232 						int size = 0;
233 						while (size == 0) {
234 							while (cursor.moveToNext()) {
235 								ids.add(cursor.getInt(cursor.getColumnIndexOrThrow(SensAppContract.Measure.ID)));
236 								long time = cursor.getLong(cursor.getColumnIndexOrThrow(SensAppContract.Measure.TIME));
237 								if (model instanceof NumericalMeasureJsonModel) {
238 									float value = cursor.getFloat(cursor.getColumnIndexOrThrow(SensAppContract.Measure.VALUE));
239 									((NumericalMeasureJsonModel) model).appendMeasure(value, time);
240 									size += INTEGER_SIZE;
241 								} else if (model instanceof StringMeasureJsonModel) {
242 									String value = cursor.getString(cursor.getColumnIndexOrThrow(SensAppContract.Measure.VALUE));
243 									((StringMeasureJsonModel) model).appendMeasure(value, time);
244 									size += value.length();
245 								}
246 								size += LONG_SIZE;
247 								if (size > sizeLimit && !cursor.isLast()) {
248 									size = 0;
249 									break;
250 								}
251 							}
252 
253 							try {
254 								RestRequest.putData(sensor.getUri(), JsonPrinter.measuresToJson(model));
255 							} catch (RequestErrorException e) {
256 								errorMessage = e.getMessage();
257 								Log.e(TAG, errorMessage);
258 								if (e.getCause() != null) {
259 									Log.e(TAG, e.getCause().getMessage());
260 								}
261 								cursor.close();
262 								return null;
263 							}
264 							model.clearValues();
265 							publishProgress((int) ((float) (progress + ids.size()) / rowTotal * 100));
266 						}
267 						ContentValues values = new ContentValues();
268 						values.put(SensAppContract.Measure.UPLOADED, 1);
269 						selection = SensAppContract.Measure.ID + " IN " + ids.toString().replace('[', '(').replace(']', ')');
270 						rowsUploaded += context.getContentResolver().update(uri, values, selection, null);
271 						progress += ids.size();
272 						ids.clear();
273 					}
274 					cursor.close();
275 				}
276 			}
277 		}
278 		return rowsUploaded;
279 	}

289 	@Override
290 	protected void onPostExecute(Integer result) {
291 		notificationManager.cancel(NOTIFICATION_ID);
292 		if (result == null) {
293 			Log.e(TAG, "Put data error");
294 			if (flag != FLAG_SILENT) {
295 				if (errorMessage == null) {
296 					Toast.makeText(context, "Upload failed", Toast.LENGTH_LONG).show();
297 				} else {
298 					Toast.makeText(context, "Upload failed:\n" + errorMessage, Toast.LENGTH_LONG).show();
299 				}
300 			}
301 		} else {
302 			Log.i(TAG, "Put data succed: " + result + " measures uploaded");
303 			if (flag != FLAG_SILENT) {
304 				final PendingIntent pendingIntent = PendingIntent.getActivity(context, 0, new Intent(context, TabsActivity.class), 0);
305 				Notification notificationFinal = new Notification(R.drawable.ic_launcher, "Upload finished", System.currentTimeMillis());
306 				notificationFinal.flags |= Notification.FLAG_AUTO_CANCEL;
307 				notificationFinal.setLatestEventInfo(context, "Upload succeed", result + " measures uploaded", pendingIntent);
308 				notificationManager.notify(NOTIFICATION_FINAL_ID, notificationFinal);
309 				if (result == 0) {
310 					Toast.makeText(context, "No measures to upload", Toast.LENGTH_LONG).show();
311 				} else {
312 					Toast.makeText(context, "Upload succeed: " + result + " measures uploaded", Toast.LENGTH_LONG).show();
313 				}
314 			}
315 		}
316 		if (listenner != null) {
317 			listenner.onTaskFinished(id);
318 		}
