180 		public static Composite getComposite(Context context, String name) {
181 			String description = null;
182 			Uri uri = null;
183 			String[] projection = {SensAppContract.Composite.DESCRIPTION, SensAppContract.Composite.URI};
184 			// Get composite description and uri
185 			Cursor cursor = context.getContentResolver().query(Uri.parse(SensAppContract.Composite.CONTENT_URI + "/" + name), projection, null, null, null);
186 			if (cursor != null) {
187 				if (cursor.getCount() > 0) {
188 					cursor.moveToFirst();
189 					description = cursor.getString(cursor.getColumnIndexOrThrow(SensAppContract.Composite.DESCRIPTION));
190 					uri = Uri.parse(cursor.getString(cursor.getColumnIndexOrThrow(SensAppContract.Composite.URI)));
191 				}
192 				cursor.close();
193 			}
194 			// Get composite sensor names
195 			ArrayList<String> sensors = new ArrayList<String>();
196 			cursor = context.getContentResolver().query(Uri.parse(SensAppContract.Sensor.CONTENT_URI + "/composite/" + name), new String[]{SensAppContract.Sensor.NAME}, null, null, null);
197 			if (cursor != null) {
198 				while (cursor.moveToNext()) {
199 					sensors.add(cursor.getString(cursor.getColumnIndexOrThrow(SensAppContract.Sensor.NAME)));
200 				}
201 				cursor.close();
202 			}
203 			// Get composite sensor addresses
204 			ArrayList<Uri> sensorUris = new ArrayList<Uri>();
205 			projection = new String[]{SensAppContract.Sensor.NAME, SensAppContract.Sensor.URI};
206 			String selection = SensAppContract.Sensor.NAME + " IN " + sensors.toString().replace("[", "(\"").replace(", ", "\", \"").replace("]", "\")");
207 			cursor = context.getContentResolver().query(SensAppContract.Sensor.CONTENT_URI, projection, selection, null, null);
208 			if (cursor != null) {
209 				while (cursor.moveToNext()) {
210 					String sensorUri = cursor.getString(cursor.getColumnIndexOrThrow(SensAppContract.Sensor.URI));
211 					sensorUri += RestRequest.SENSOR_PATH + "/" + cursor.getString(cursor.getColumnIndexOrThrow(SensAppContract.Sensor.NAME));
212 					sensorUris.add(Uri.parse(sensorUri));
213 				}
214 				cursor.close();
215 			}
216 			return new Composite(name, description, uri, sensorUris);
217 		}
