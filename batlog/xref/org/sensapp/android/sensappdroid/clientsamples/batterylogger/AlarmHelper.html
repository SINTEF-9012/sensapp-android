16  package org.sensapp.android.sensappdroid.clientsamples.batterylogger;
17  
18  import android.app.AlarmManager;
19  import android.app.Notification;
20  import android.app.NotificationManager;
21  import android.app.PendingIntent;
22  import android.content.Context;
23  import android.content.Intent;
24  
25  public final class AlarmHelper {
26  
27  	protected static final int REFRESH_RATE = 60; // Polling interval in seconds to start the service.
28  	
29  	private static final int ACTIVE_NOTIFICATION_ID = 79290;
30  	
31  	private AlarmHelper() {}
32  	
33  	protected static void setAlarm(Context context) {
34  		Intent startService = new Intent(context, BatteryLoggerService.class);
35  		PendingIntent pendingIntent = PendingIntent.getService(context, 0, startService, PendingIntent.FLAG_CANCEL_CURRENT);
36  		((AlarmManager) context.getSystemService(Context.ALARM_SERVICE)).setRepeating(AlarmManager.RTC_WAKEUP, System.currentTimeMillis(), REFRESH_RATE * 1000, pendingIntent);
37  		
38  		@SuppressWarnings("deprecation") // We don't want target only API 16...
39  		Notification notification = new Notification.Builder(context)
40  			.setContentTitle("BatLog active")
41  			.setContentText("BatLog active")
42  		    .setSmallIcon(R.drawable.ic_launcher)
43  		    .setContentIntent(PendingIntent.getActivity(context, 0, new Intent(context, MainActivity.class), 0))
44  		    .getNotification();
45  		notification.flags = notification.flags | Notification.FLAG_ONGOING_EVENT;
46  		((NotificationManager) context.getSystemService(Context.NOTIFICATION_SERVICE)).notify(ACTIVE_NOTIFICATION_ID, notification);
47  	}
48  	
49  	protected static void cancelAlarm(Context context) {
50  		Intent startService = new Intent(context, BatteryLoggerService.class);
51  		PendingIntent pendingIntent = PendingIntent.getService(context, 0, startService, PendingIntent.FLAG_CANCEL_CURRENT);
52  		((AlarmManager) context.getSystemService(Context.ALARM_SERVICE)).cancel(pendingIntent);
53  		
54  		((NotificationManager) context.getSystemService(Context.NOTIFICATION_SERVICE)).cancel(ACTIVE_NOTIFICATION_ID);
55  	}
56  }
