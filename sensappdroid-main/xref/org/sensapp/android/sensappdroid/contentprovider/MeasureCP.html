56  	@Override
57  	public Cursor query(Uri uri, String[] projection, String selection, String[] selectionArgs, String sortOrder, int uid) throws IllegalStateException {
58  		checkColumns(projection);
59  		SQLiteQueryBuilder queryBuilder = new SQLiteQueryBuilder();
60  		queryBuilder.setTables(MeasureTable.TABLE_MEASURE);
61  		switch (measureURIMatcher.match(uri)) {
62  		case MEASURES:
63  			if (!isSensAppUID(uid)) {
64  				throw new IllegalStateException("Forbiden uri");
65  			}
66  			break;
67  		case MEASURE_ID:
68  			if (!isSensAppUID(uid)) {
69  				throw new IllegalStateException("Forbiden uri");
70  			}
71  			queryBuilder.appendWhere(MeasureTable.COLUMN_ID + "=" + uri.getLastPathSegment());
72  			break;
73  		case MEASURE_SENSOR:
74  			queryBuilder.appendWhere(MeasureTable.COLUMN_SENSOR + "= \"" + uri.getLastPathSegment() + "\"");
75  			break;
76  		case MEASURE_COMPOSITE:
77  			if (!isSensAppUID(uid)) {
78  				throw new IllegalStateException("Forbiden uri");
79  			}
80  			Cursor c = getContext().getContentResolver().query(Uri.parse(SensAppContract.Sensor.CONTENT_URI + "/composite/" + uri.getLastPathSegment()), new String[]{SensorTable.COLUMN_NAME}, null, null, null);
81  			ArrayList<String> names = new ArrayList<String>();
82  			if (c != null) {
83  					while (c.moveToNext()) {
84  						names.add(c.getString(c.getColumnIndexOrThrow(SensorTable.COLUMN_NAME)));
85  				}
86  				c.close();
87  			} else {
88  				return null;
89  			}
90  			queryBuilder.appendWhere(MeasureTable.COLUMN_SENSOR + " IN " + names.toString().replace("[", "(\"").replace(", ", "\", \"").replace("]", "\")"));
91  			break;
92  		default:
93  			throw new IllegalArgumentException("Unknown URI: " + uri);
94  		}
95  		SQLiteDatabase db = getDatabase().getWritableDatabase();
96  		Cursor cursor = queryBuilder.query(db, projection, selection, selectionArgs, null, null, sortOrder);
97  		cursor.setNotificationUri(getContext().getContentResolver(), uri);
98  		return cursor;
99  	}
