45  	private static final UriMatcher sensorURIMatcher = new UriMatcher(UriMatcher.NO_MATCH);
46  	static {

49  		sensorURIMatcher.addURI(SensAppContentProvider.AUTHORITY, BASE_PATH + "/uid/*", SENSOR_UID);
50  		sensorURIMatcher.addURI(SensAppContentProvider.AUTHORITY, BASE_PATH + "/*", SENSOR_ID);
51  	}

57  	@Override
58  	public Cursor query(Uri uri, String[] projection, String selection, String[] selectionArgs, String sortOrder, int uid) throws IllegalStateException {
59  		checkColumns(projection);
60  		SQLiteQueryBuilder queryBuilder = new SQLiteQueryBuilder();
61  		switch (sensorURIMatcher.match(uri)) {
62  		case SENSORS:
63  			if (!isSensAppUID(uid)) {
64  				throw new IllegalStateException("Forbiden uri");
65  			}
66  			queryBuilder.setTables(SensorTable.TABLE_SENSOR);
67  			break;
68  		case COMPOSITE_SENSORS:
69  			if (!isSensAppUID(uid)) {
70  				throw new IllegalStateException("Forbiden uri");
71  			}
72  			Hashtable<String, String> columnMap = new Hashtable<String, String>();
73  			columnMap.put(SensorTable.COLUMN_NAME, SensorTable.TABLE_SENSOR + "." + SensorTable.COLUMN_NAME);
74  			columnMap.put(SensorTable.COLUMN_BACKEND, SensorTable.TABLE_SENSOR + "." + SensorTable.COLUMN_BACKEND);
75  			columnMap.put(SensorTable.COLUMN_DESCRIPTION, SensorTable.TABLE_SENSOR + "." + SensorTable.COLUMN_DESCRIPTION);
76  			columnMap.put(SensorTable.COLUMN_TEMPLATE, SensorTable.TABLE_SENSOR + "." + SensorTable.COLUMN_TEMPLATE);
77  			columnMap.put(SensorTable.COLUMN_UNIT, SensorTable.TABLE_SENSOR + "." + SensorTable.COLUMN_UNIT);
78  			columnMap.put(SensorTable.COLUMN_UPLOADED, SensorTable.TABLE_SENSOR + "." + SensorTable.COLUMN_UPLOADED);
79  			columnMap.put(SensorTable.COLUMN_URI, SensorTable.TABLE_SENSOR + "." + SensorTable.COLUMN_URI);
80  			columnMap.put(SensorTable.COLUMN_ICON, SensorTable.TABLE_SENSOR + "." + SensorTable.COLUMN_ICON);
81  			columnMap.put(SensorTable.COLUMN_CLIENT_UID, SensorTable.TABLE_SENSOR + "." + SensorTable.COLUMN_CLIENT_UID);
82  			queryBuilder.setProjectionMap(columnMap);
83  			queryBuilder.setTables(SensorTable.TABLE_SENSOR + ", " + ComposeTable.TABLE_COMPOSE);
84  			queryBuilder.appendWhere(SensorTable.TABLE_SENSOR + "." + SensorTable.COLUMN_NAME + " = " + ComposeTable.COLUMN_SENSOR
85  					 + " AND " + ComposeTable.COLUMN_COMPOSITE + " = \"" + uri.getLastPathSegment() + "\"");
86  			break;
87  		case SENSOR_UID:
88  			queryBuilder.setTables(SensorTable.TABLE_SENSOR);
89  			queryBuilder.appendWhere(SensorTable.COLUMN_NAME + " = \"" + uri.getLastPathSegment() + "\"");
90  			SQLiteDatabase db = getDatabase().getWritableDatabase();
91  			Cursor cursor = queryBuilder.query(db, new String[]{SensorTable.COLUMN_CLIENT_UID}, null, null, null, null, null);
92  			cursor.setNotificationUri(getContext().getContentResolver(), uri);
93  			return cursor;
94  		case SENSOR_ID:
95  			int sensorUID = getSensorUID(uri.getLastPathSegment());
96  			if (!isSensAppUID(uid) && sensorUID != 0 && sensorUID != uid) {
97  				throw new IllegalStateException("Forbiden uri");
98  			}
99  			queryBuilder.setTables(SensorTable.TABLE_SENSOR);
100 			queryBuilder.appendWhere(SensorTable.COLUMN_NAME + " = \"" + uri.getLastPathSegment() + "\"");
101 			break;
102 		default:
103 			throw new IllegalArgumentException("Unknown URI: " + uri);
104 		}
105 		SQLiteDatabase db = getDatabase().getWritableDatabase();
106 		Cursor cursor = queryBuilder.query(db, projection, selection, selectionArgs, null, null, sortOrder);
107 		cursor.setNotificationUri(getContext().getContentResolver(), uri);
108 		return cursor;
109 	}

111 	@Override
112 	public Uri insert(Uri uri, ContentValues values, int uid) {
113 		SQLiteDatabase db = getDatabase().getWritableDatabase();
114 		long id = 0;
115 		switch (sensorURIMatcher.match(uri)) {
116 		case SENSORS:
117 			values.put(SensorTable.COLUMN_UPLOADED, 0);
118 			values.put(SensorTable.COLUMN_CLIENT_UID, uid);
119 			id = db.insert(SensorTable.TABLE_SENSOR, null, values);
120 			break;
121 		default:
122 			throw new IllegalArgumentException("Bad URI: " + uri);
123 		}
124 		getContext().getContentResolver().notifyChange(uri, null);
125 		return Uri.parse("content://" + SensAppContentProvider.AUTHORITY + "/" + BASE_PATH + "/" + id);
126 	}

128 	@Override
129 	public int delete(Uri uri, String selection, String[] selectionArgs, int uid) throws IllegalStateException {
130 		SQLiteDatabase db = getDatabase().getWritableDatabase();
131 		int rowsDeleted = 0;
132 		switch (sensorURIMatcher.match(uri)) {
133 		case SENSORS:
134 			if (!isSensAppUID(uid)) {
135 				throw new IllegalStateException("Forbiden uri");
136 			}
137 			rowsDeleted = db.delete(SensorTable.TABLE_SENSOR, selection, selectionArgs);
138 			// Clean composites
139 			db.delete(ComposeTable.TABLE_COMPOSE, null, null);
140 			break;
141 		case SENSOR_ID:
142 			String name = uri.getLastPathSegment();
143 			if (!isSensAppUID(uid) && getSensorUID(name) != uid) {
144 				throw new IllegalStateException("Forbiden uri");
145 			}
146 			if (TextUtils.isEmpty(selection)) {
147 				rowsDeleted = db.delete(SensorTable.TABLE_SENSOR, SensorTable.COLUMN_NAME + " = \"" + name + "\"", null);
148 			} else {
149 				rowsDeleted = db.delete(SensorTable.TABLE_SENSOR, SensorTable.COLUMN_NAME + " = \"" + name + "\" and " + selection, selectionArgs);
150 			}
151 			// Clean composites
152 			db.delete(ComposeTable.TABLE_COMPOSE, ComposeTable.COLUMN_SENSOR + " = \"" + name + "\"", null);
153 			break;
154 		default:
155 			throw new IllegalArgumentException("Unknown URI: " + uri);
156 		}
157 		getContext().getContentResolver().notifyChange(uri, null);
158 		getContext().getContentResolver().notifyChange(Uri.parse("content://" + SensAppContentProvider.AUTHORITY + "/" + BASE_PATH + "/composite"), null);
159 		return rowsDeleted;
160 	}
