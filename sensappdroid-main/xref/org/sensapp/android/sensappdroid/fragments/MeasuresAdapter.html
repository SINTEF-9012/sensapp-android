16  package org.sensapp.android.sensappdroid.fragments;
17  
18  import java.util.Hashtable;
19  
20  import org.sensapp.android.sensappdroid.R;
21  import org.sensapp.android.sensappdroid.contract.SensAppContract;
22  
23  import android.content.Context;
24  import android.database.Cursor;
25  import android.graphics.BitmapFactory;
26  import android.view.LayoutInflater;
27  import android.view.View;
28  import android.view.ViewGroup;
29  import android.widget.CursorAdapter;
30  import android.widget.ImageView;
31  import android.widget.TextView;
32  
33  public class MeasuresAdapter extends CursorAdapter {
34  
35  	private Hashtable<String, byte[]> icons =  new Hashtable<String, byte[]>();
36  	
37  	private LayoutInflater inflater;
38  	private int indexSensor;
39  	private int indexValue;
40  	
41  	public MeasuresAdapter(Context context, Cursor c) {
42  		super(context, c, 0);
43  		inflater = LayoutInflater.from(context);
44  	}
45  	
46  	public Hashtable<String, byte[]> getIcons() {
47  		return icons;
48  	}
49  	
50  	@Override
51  	public Cursor swapCursor(Cursor newCursor) {
52  		if (newCursor != null) {
53  			indexSensor = newCursor.getColumnIndex(SensAppContract.Measure.SENSOR);
54  			indexValue = newCursor.getColumnIndex(SensAppContract.Measure.VALUE);
55  		}
56  		return super.swapCursor(newCursor);
57  	}
58  
59  	@Override
60  	public void bindView(View view, Context context, Cursor cursor) {
61  		ImageView iconIv = (ImageView) view.findViewById(R.id.icon);
62  		TextView value = (TextView) view.findViewById(R.id.value);
63  		
64  		// Retrieve the icon if exits
65  		String sensor = cursor.getString(indexSensor);
66  		byte[] icon = icons.get(sensor); 
67  		if (icon != null) {
68  			iconIv.setImageBitmap(BitmapFactory.decodeByteArray(icon, 0, icon.length));
69  		}
70  		
71  		String valueString = cursor.getString(indexValue);
72  		// Take care of large values concatenation
73  		if (valueString.length() > 10) {
74  			valueString = valueString.substring(0, 6) + "...";
75  		}
76  		value.setText(valueString);
77  	}
78  	
79  	@Override
80  	public View newView(Context context, Cursor cursor, ViewGroup parent) {
81  		return inflater.inflate(R.layout.measure_row, parent, false);
82  	}
83  }
