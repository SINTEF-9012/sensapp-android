16  package org.sensapp.android.sensappdroid.preferences;
17  
18  import org.sensapp.android.sensappdroid.R;
19  
20  import android.app.AlertDialog;
21  import android.content.Context;
22  import android.content.DialogInterface;
23  import android.preference.EditTextPreference;
24  import android.util.AttributeSet;
25  import android.util.Patterns;
26  import android.webkit.URLUtil;
27  
28  public class EditTextServerPreference extends EditTextPreference {
29  
30  	private String lastValid;
31  	
32  	public EditTextServerPreference(Context context) {
33  		super(context);
34  		lastValid = context.getString(R.string.pref_server_default_value);
35  	}
36  
37  	public EditTextServerPreference(Context context, AttributeSet attrs) {
38  		super(context, attrs);
39  		lastValid = context.getString(R.string.pref_server_default_value);
40  	}
41  
42  	public EditTextServerPreference(Context context, AttributeSet attrs, int defStyle) {
43  		super(context, attrs, defStyle);
44  		lastValid = context.getString(R.string.pref_server_default_value);
45  	}
46  	
47  	@Override
48  	protected void onDialogClosed(boolean positiveResult) {
49  		super.onDialogClosed(positiveResult);
50  		if (positiveResult) {
51  			String address = this.getText();
52  			boolean valid = true;
53  			String errorMsg = "";
54  			if (!Patterns.WEB_URL.matcher(address).matches()) {
55  				valid = false;
56  				errorMsg += "The server url is not a valid web address. Check for unexpected spaces or characters.";
57  			} else if (!URLUtil.isValidUrl(address)) {
58  				address = "http://".concat(address);
59  			} else if (!URLUtil.isHttpUrl(address)) {
60  				valid = false;
61  				errorMsg += "The server url is not a HTTP one. At this time only this protocol is supported.";
62  			}
63  			if (!valid) {
64  				AlertDialog.Builder builder = new AlertDialog.Builder(getContext());
65  				builder.setMessage(errorMsg)
66  				.setCancelable(false)
67  				.setPositiveButton("Ok", new DialogInterface.OnClickListener() {
68  					public void onClick(DialogInterface dialog, int id) {
69  						showDialog(null);
70  					}
71  				}).create().show();
72  			} else {
73  				this.setText(address);
74  				lastValid = address;
75  			}
76  		} else {
77  			this.setText(lastValid);
78  		}
79  	}
80  }
