16  package org.sensapp.android.sensappdroid.preferences;
17  
18  import java.util.HashSet;
19  import java.util.Set;
20  
21  import org.sensapp.android.sensappdroid.contract.SensAppContract;
22  
23  import android.app.AlertDialog;
24  import android.app.Dialog;
25  import android.app.DialogFragment;
26  import android.content.DialogInterface;
27  import android.database.Cursor;
28  import android.os.Bundle;
29  import android.preference.PreferenceManager;
30  
31  public class AutoUploadSensorDialog extends DialogFragment {
32  
33  	public static final String SENSOR_MAINTAINED = "sensor_maintained_key";
34  	
35  	private Cursor cursor;
36  	
37      @Override
38      public Dialog onCreateDialog(Bundle savedInstanceState) {
39  		cursor = getActivity().getContentResolver().query(SensAppContract.Sensor.CONTENT_URI, null, null, null, null);
40  		String[] names = new String[cursor.getCount()];
41  		boolean[] status = new boolean[cursor.getCount()];
42  		final Set<String> statusSaved = PreferenceManager.getDefaultSharedPreferences(getActivity()).getStringSet(SENSOR_MAINTAINED, new HashSet<String>());
43  		for (int i = 0 ; cursor.moveToNext() ; i ++) {
44  			names[i] = cursor.getString(cursor.getColumnIndexOrThrow(SensAppContract.Sensor.NAME));
45  			status[i] = statusSaved.contains(names[i]);
46  		}
47  		return new AlertDialog.Builder(getActivity())
48  		.setTitle("Select sensors to auto upload")
49  		.setMultiChoiceItems(names, status, new DialogInterface.OnMultiChoiceClickListener() {
50  			public void onClick(DialogInterface dialog, int which, boolean isChecked) {
51  				cursor.moveToPosition(which);
52  				String sensorName = cursor.getString(cursor.getColumnIndexOrThrow(SensAppContract.Sensor.NAME));
53  				if (isChecked) {
54  					statusSaved.add(sensorName);
55  				} else {
56  					statusSaved.remove(sensorName);
57  				}
58  			}
59  		})
60  		.setPositiveButton("Done", new DialogInterface.OnClickListener() {
61  			public void onClick(DialogInterface dialog, int whichButton) {
62  				PreferenceManager.getDefaultSharedPreferences(getActivity()).edit().putStringSet(SENSOR_MAINTAINED, statusSaved).commit();
63  				cursor.close();
64  			}
65  		}).create();
66      }
67  
68  	@Override
69  	public void onCancel(DialogInterface dialog) {
70  		super.onCancel(dialog);
71  		if (cursor != null) {
72  			cursor.close();
73  		}
74  	}    
75  }
